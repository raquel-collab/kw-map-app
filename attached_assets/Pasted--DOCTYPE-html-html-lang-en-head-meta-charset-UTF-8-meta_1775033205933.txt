let selectedSet = new Set();
function toggleSelect(i, forceOn){
  if(forceOn===true){ selectedSet.add(i); }
  else if(forceOn===false){ selectedSet.delete(i); }
  else { if(selectedSet.has(i)) selectedSet.delete(i); else selectedSet.add(i); }

  const chk=document.getElementById('chk'+i);
  const on=selectedSet.has(i);
  card.classList.toggle('selected',on);
  chk.classList.toggle('on',on);
  updateSelCount();
}
function selectAll(on){ step1Pages.forEach((_,i)=>toggleSelect(i,on)); }
function updateSelCount(){ document.getElementById('sel-count').textContent=selectedSet.size+' selected'; }

  navigator.clipboard.writeText(txt);
}
